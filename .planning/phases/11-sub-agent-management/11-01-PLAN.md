    ```typescript
    /**
     * Reparent an agent: change its parent_agent_id and optionally its department_id.
     * Used by drag-and-drop in the agent tree view.
     *
     * Rules:
     * 1. Cannot reparent to self
     * 2. Cannot create circular reference (agent cannot become child of its own descendant)
     * 3. Must be same business
     * 4. If new parent is in a different department, update department_id too
     * 5. If dropping on a department (no parent), set parent_agent_id = null and department_id = target dept
     */
    export async function reparentAgent(
      supabase: SupabaseClient,
      agentId: string,
      businessId: string,
      newParentAgentId: string | null,
      newDepartmentId: string,
    ): Promise<void>
    ```

    Implementation:
    1. Fetch the agent being moved: `select("id, name, parent_agent_id, department_id").eq("id", agentId).eq("business_id", businessId).single()`
    2. If `newParentAgentId` is not null:
       a. Check it's not the same as agentId (self-reference)
       b. Fetch the target parent agent to verify it's in the same business
       c. Check for circular reference: walk up the parent chain from newParentAgentId. If any ancestor's id === agentId, throw "Cannot reparent: would create circular reference"
       d. Get the target parent's department_id and use it as the newDepartmentId (sub-agent inherits parent's department)
    3. Update the agent: `update({ parent_agent_id: newParentAgentId, department_id: newDepartmentId }).eq("id", agentId).eq("business_id", businessId)`
    4. If the agent was a lead with children and is now being moved to a different department, move children too:
       `update({ department_id: newDepartmentId }).eq("parent_agent_id", agentId).eq("business_id", businessId)`
    5. Create audit log (best-effort):
       ```
       business_id: businessId,
       action: "agent.reparented",
       entity_type: "agent",
       entity_id: agentId,
       metadata: { previous_parent: agent.parent_agent_id, new_parent: newParentAgentId, previous_department: agent.department_id, new_department: newDepartmentId }
       ```

    The circular reference check function (private helper):
    ```typescript
    async function wouldCreateCycle(
      supabase: SupabaseClient,
      agentId: string,
      targetParentId: string,
      businessId: string,
    ): Promise<boolean> {
      let currentId: string | null = targetParentId;
      const visited = new Set<string>();
      while (currentId) {
        if (currentId === agentId) return true;
        if (visited.has(currentId)) return false; // safety: break infinite loop
        visited.add(currentId);
        const { data } = await supabase
          .from("agents")
          .select("parent_agent_id")
          .eq("id", currentId)
          .eq("business_id", businessId)
          .single();
        currentId = (data?.parent_agent_id as string) ?? null;
      }
      return false;
    }
    ```

    Add `reparentAgentAction` server action:

    ```typescript
    /**
     * Reparent an agent (move to a different parent or department).
     * Used by drag-and-drop in the agent tree view.
     */
    export async function reparentAgentAction(
      agentId: string,
      businessId: string,
      newParentAgentId: string | null,
      newDepartmentId: string,
    ) {
      const supabase = await createServerClient();
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) { redirect("/sign-in"); }

      try {
        const { reparentAgent } = await import("@agency-factory/core/server");
        await reparentAgent(supabase, agentId, businessId, newParentAgentId, newDepartmentId);
      } catch (err) {
        return { error: err instanceof Error ? err.message : "Failed to reparent agent" };
      }

      revalidatePath(`/businesses/${businessId}/agents`);
    }
    ```
