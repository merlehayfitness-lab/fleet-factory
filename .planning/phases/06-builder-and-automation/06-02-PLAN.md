Wire the deployment pipeline to push OpenClaw workspaces to the VPS, add health checks, real-time deployment streaming, and deployment UI components.

Purpose: This plan evolves the existing deployment service to actually push artifacts to the VPS via REST API. Health checks verify each agent is responding after deployment.

</objective>

    **Serial deployment queuing guard (per locked decision):**
    Before starting the VPS push (and ideally before generating artifacts), add a guard:

    ```typescript
    // SERIAL DEPLOYMENT GUARD — Per locked decision: "Deployments processed serially (queued)"
    // Check for any active VPS deployment for this business before proceeding.
    const { data: activeDeployments } = await supabase
      .from("deployments")
      .select("id, status")
      .eq("business_id", businessId)
      .eq("deploy_target", "vps")
      .in("status", ["building", "deploying", "verifying"])
      .limit(1);

    if (activeDeployments && activeDeployments.length > 0) {
      // Another deployment is in progress — keep this one queued
      // Update status message so the UI can inform the user
      await supabase.from("deployments").update({
        status: "queued",
        config_snapshot: {
          ...(existingSnapshot ?? {}),
          queue_reason: `Waiting for deployment ${activeDeployments[0].id} to complete`,
        },
      }).eq("id", deploymentId);

      return {
        deploymentId,
        status: "queued",
        message: `Deployment queued — waiting for active deployment ${activeDeployments[0].id} to finish`,
      };
    }
    ```

    After updating deployment with artifacts, add VPS push:

    ```
    // 10b. If VPS is configured, push to VPS
    if (isVpsConfigured()) {
      // Update deploy_target to 'vps'
      await supabase.from("deployments").update({ deploy_target: "vps" }).eq("id", deploymentId);

      // Build VPS payload from workspace
      const vpsPayload: VpsDeployPayload = {
        businessId,
        businessSlug: business.slug,
        deploymentId,
        version: nextVersion,
        isRollback: false,
        skipOptimization: false,
        agents: agents mapped to { agentId, vpsAgentId, departmentType, model },
        workspaceFiles: openclawWorkspace.files,
        openclawConfig: openclawWorkspace.config,
      };

      const vpsResult = await pushDeploymentToVps(supabase, deploymentId, vpsPayload);

      if (!vpsResult.success) {
        // VPS push failed but artifacts are generated -- mark as failed
        throw new Error(`VPS deployment failed: ${vpsResult.error}`);
      }

      // Store optimization report
      if (vpsResult.optimizationReport) {
        await supabase.from("deployments").update({
          optimization_report: vpsResult.optimizationReport,
        }).eq("id", deploymentId);
      }

      // 10c. Transition to 'verifying' and run post-deploy health check
      assertDeploymentTransition("deploying", "verifying");
      await updateDeploymentStatus(supabase, deploymentId, "verifying");

      const healthResult = await runPostDeployHealthCheck(supabase, businessId, agents);

      if (healthResult.unhealthy.length > 0 && healthResult.healthy.length > 0) {
        // Partial failure: some agents deployed, some failed
        // Store partial results, still transition to live (degraded behavior)
        // The dashboard will show which agents failed
      }

      if (healthResult.healthy.length === 0) {
        throw new Error("Post-deploy health check failed: no agents responding");
      }
    }
    // If VPS not configured, continue with existing local-only path
    ```
